package com.mercedesbenz.carenquiry.model;

import com.fasterxml.jackson.annotation.JsonIgnore;
import jakarta.persistence.*;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Positive;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.autoconfigure.web.WebProperties;

import javax.annotation.processing.Generated;

//to Create a Database Schema
@Entity
@Table(name = "car")
public class Cars {

    /**
     * To create a table named cars which has a PK id, column namely carname,carimage and carprice
     */
    @Id
//    @JsonIgnore
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private  Integer id ;
   @Column   @Value("string value")
    private  String carname;
    @Column  @Value("string value")
    private  String carimage;
    @Column
    private  int carprice;

    public Cars(String carname, String carimage, int carprice) {
        this.carname = carname;
        this.carimage = carimage;
        this.carprice = carprice;
    }
    public Cars(Integer id,String carname, String carimage, int carprice) {
        this.id=id;
        this.carname = carname;
        this.carimage = carimage;
        this.carprice = carprice;
    }
public  Cars(){

}

    public Integer getId() {
        return id;
    }

    public void setId(Integer id) {
        this.id = id;
    }
    public String getCarimage() {
        return carimage;
    }

    public void setCarimage(String carimage) {
        this.carimage = carimage;
    }

    public int getCarprice() {
        return carprice;
    }

    public void setCarprice(int carprice) {
        this.carprice = carprice;
    }

    public String getCarname() {
        return carname;
    }

    public void setCarname(String carname) {
        this.carname = carname;
    }

}
